import Ogp from "./partials/ogp.mdx";
import Header from "./partials/header.mdx";
import Footer from "./partials/footer.mdx";

<html lang="ja">
  <head>
    <meta charset="utf-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <Ogp {...props} />
    <link rel="stylesheet" href="/main.css" />
    <title>
      {props.title && `${props.title} | `}
      {props.site.name}
    </title>
    <script
      type="importmap"
      dangerouslySetInnerHTML={{
        __html: `
      {
        "imports": {
          "is-land": "/is-land.js",
          "preact": "https://esm.sh/preact",
          "preact/hooks": "https://esm.sh/preact/hooks?external=preact",
          "preact/jsx-runtime": "https://esm.sh/preact/jsx-runtime?external=preact"
        }
      }
    `,
      }}
    />
    {process.env.NODE_ENV === "development" && (
      <script
        dangerouslySetInnerHTML={{
          __html: `
        // Hooking WebSockets to rehydrate after a hot reload
        (function() {
          const OriginalWebSocket = window.WebSocket;
          window.WebSocket = function(url, protocols) {
            const ws = new OriginalWebSocket(url, protocols);
            ws.addEventListener("message", (event) => {
              try {
                const data = JSON.parse(event.data);
                if (data.type === "eleventy.reload") {
                  // Schedule rehydration after morphdom completes
                  requestAnimationFrame(() => {
                    window.__eleventyRehydrate?.();
                  });
                }
              } catch (e) { console.error(e) }
            });
            return ws;
          };
          window.WebSocket.prototype = OriginalWebSocket.prototype;
          window.WebSocket.CONNECTING = OriginalWebSocket.CONNECTING;
          window.WebSocket.OPEN = OriginalWebSocket.OPEN;
          window.WebSocket.CLOSING = OriginalWebSocket.CLOSING;
          window.WebSocket.CLOSED = OriginalWebSocket.CLOSED;
        })();
      `,
        }}
      />
    )}
    <script
      type="module"
      dangerouslySetInnerHTML={{
        __html: `
      import { Island } from "is-land";
      import { h, hydrate } from "preact";
      Island.attributePrefix = "land-on:";

      const mount = async (target) => {
        const component = await import(target.getAttribute("import"));
        const props = JSON.parse(target.getAttribute("props"));
        hydrate(h(component.default, props), target);
      };

      Island.addInitType("preact", mount);

      ${process.env.NODE_ENV === "development" ? `
      // Rehydrate after hot reload
      window.__eleventyRehydrate = () => {
        document.querySelectorAll('is-land[type="preact"]').forEach(el => {
          if (el.hasAttribute("import")) {
            mount(el);
          }
        });
      };
      ` : ""}
    `,
      }}
    />
  </head>
  <body>
    <Header {...props} />
    <div
      class="mx-auto mb-4 min-h-[60vh] max-w-6xl px-4"
      dangerouslySetInnerHTML={{ __html: props.content }}
    ></div>
    <Footer {...props} />
  </body>
</html>
