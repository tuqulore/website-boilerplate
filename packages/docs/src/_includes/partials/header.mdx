import { Island } from "@tuqulore-inc/eleventy-preset/island";
import { eleventy } from "@tuqulore-inc/eleventy-preset/eleventy";
import MobileMenu from "./mobile-menu.client.jsx";

export function switchLocale(url, target) {
  const match = url.match(/^\/(ja|en)(\/.*)?$/);
  return match ? `/${target}${match[2] ?? "/"}` : `/${target}/`;
}

<header class="sticky top-0 z-40 mb-8 bg-white shadow">
  <div class="mx-auto flex h-16 max-w-6xl items-center gap-4 px-4">
    <a class="flex-shrink-0" href="/">
      {eleventy.site.name}
    </a>
    <div class="ml-auto flex items-center gap-4">
      <nav
        aria-label={eleventy.locale === "en" ? "Language" : "言語"}
        class="flex gap-4"
      >
        <a href={switchLocale(eleventy.page.url, "ja")}>日本語</a>
        <a href={switchLocale(eleventy.page.url, "en")}>English</a>
      </nav>
      <Island
        component={MobileMenu}
        on="interaction"
        class="md:hidden"
        locale={eleventy.locale ?? "ja"}
        currentUrl={eleventy.page.url}
        nav={eleventy.nav[eleventy.locale ?? "ja"] ?? []}
      />
    </div>
  </div>
</header>
