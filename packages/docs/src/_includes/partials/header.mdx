import { eleventy } from "@tuqulore-inc/eleventy-preset/eleventy";

export function switchLocale(url, target) {
  const match = url.match(/^\/(ja|en)(\/.*)?$/);
  return match ? `/${target}${match[2] ?? "/"}` : `/${target}/`;
}

<header class="sticky top-0 mb-8 bg-white shadow">
  <div class="mx-auto flex h-16 max-w-6xl items-center gap-4 px-4">
    <a class="flex-shrink-0" href="/">
      {eleventy.site[eleventy.locale ?? "ja"].name}
    </a>
    <nav class="ml-auto flex gap-4">
      <a href={switchLocale(eleventy.page.url, "ja")}>日本語</a>
      <a href={switchLocale(eleventy.page.url, "en")}>English</a>
    </nav>
  </div>
</header>
